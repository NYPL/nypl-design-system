export const makeColorCard = function (data) {
  const { colorSource, dataBlack, dataWhite, name, textColor } = data;
  const card = (
    <ColorCard
      backgroundColor={name}
      colorName={name}
      colorSource={colorSource}
      dataBlack={dataBlack}
      dataWhite={dataWhite}
      textColor={textColor}
      key={name}
    />
  );
  return card;
};
export const getColorCards = (category) => {
  const cards = [];
  const catArr = cssVars[category];
  for (let i = 0; i < catArr.length; i++) {
    const colorData = catArr[i];
    const card = makeColorCard(colorData);
    cards.push(card);
  }
  return cards;
};

## Overview

The Reservoir Design System (DS) the provides the NYPL branded and approved
color palette. The NYPL color palette is used to style the DS components and
is also available for consuming applications in the form of javascript and
CSS design tokens. Please refer to the [Design Tokens](http://localhost:6006/?path=/story/style-guide-design-tokens--page)
page for information about consuming and using all style tokens.

Whenever using these colors, designer must ensure that the foreground and
background colors pass, at a minimum, the `WCAG AA` accessibility guidelines for
color contrast. For reference, to meet the `WCAG AA` standard, the contrast ratio
must be at least 4.5:1. To meet the `WCAG AAA` standard, the contrast ratio must
be at least 7:1.
