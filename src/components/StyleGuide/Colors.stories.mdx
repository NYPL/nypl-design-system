export const makeColorCard = function (data) {
  const name = data.name;
  const dataBlack = data.dataBlack;
  const dataWhite = data.dataWhite;
  const textColor = data.textColor;
  const card = (
    <ColorCard
      backgroundColor={name}
      colorName={name}
      dataBlack={dataBlack}
      dataWhite={dataWhite}
      textColor={textColor}
      key={name}
    />
  );
  return card;
};
export const getColorCards = (cat) => {
  const cards = [];
  const catArr = cssVars[cat];
  for (let i = 0; i < catArr.length; i++) {
    const colorData = catArr[i];
    const card = makeColorCard(colorData);
    cards.push(card);
  }
  return cards;
};
