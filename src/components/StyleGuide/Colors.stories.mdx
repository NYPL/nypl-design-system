  "--nypl-colors-ui-gray-xdark",

export const uiVarDocs = [];
export const grayscaleVarDocs = [];
export const makeColorCard = function (key, value, docArray) {
  docArray.push(
    <ColorCard
      backgroundColor={value}
      foregroundColor="#ffffff"
      colorName={key}
    />
  );
};
for (let i = 0; i < uiVars_arr.length; i++) {
  const varName = uiVars_arr[i];
  const varValue = getCSSVariable(varName);
  makeColorCard(varName, varValue, uiVarDocs);
}
for (let i = 0; i < grayscaleVars_arr.length; i++) {
  const varName = grayscaleVars_arr[i];
  const varValue = getCSSVariable(varName);
  makeColorCard(varName, varValue, grayscaleVarDocs);
}
